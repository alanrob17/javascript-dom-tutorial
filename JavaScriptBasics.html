Note:

Originally we set up the function like this.

 calcAge: function(birthYear) {
  return 2019 - this.birthYear;
 }

We then had to calculate the age by.

 console.log(john.calcAge(john.birthYear)); // 29

Fortunately we can improve this code.

    this.birthYear

will get the current birthYear in the Object literal so we don't have to calculate it ourselves.

If we dump the Object with,

    console.log(john);

We won't get the calculated age. calcAge will return,

    calcAge: [Function: calcAge]

We can work around this by creating a variable in the function.

    calcAge: function(birthYear) {
      this.age = 2019 - this.birthYear;
    }

Now we can access the calculated age.

    john.calcAge(); // have to do this
    console.log(john);

Also note that we have to run the calcAge function before we can access the .age property.

  2. Create a function (not a method) to calculate the average of a given array of tips. HINT: Loop over the array, and in each iteration store the current sum in a variable (starting from 0).   3. Calculate the average tip for each family
  4. Log to the console which family paid the highest tips on average

.

    var john = {
        fullName: 'John Smith',
        bills: [124, 48, 268, 180, 42],
        calcTips: function() {
         this.tips = [];
         this.finalValues = [];

         for (var i = 0; i < this.bills.length; i++) {
            // Determine percentage based on tipping rules
            var percentage;
            var bill = this.bills[i];

            if (bill < 50) {
                percentage = 0.2;
            } else if (bill >= 50 && bill < 200) {
                percentage = 0.15;
            } else {
                percentage = 0.1;
            }

            // Add results to the corresponding arrays
            this.tips[i] = bill * percentage;
            this.finalValues[i] = bill + bill * percentage;
         }
        }
    };

    var mark = {
        fullName: 'Mark Miller',
        bills: [77, 475, 110, 45],
        calcTips: function() {
            this.tips = [];
            this.finalValues = [];
            
            for (var i = 0; i < this.bills.length; i++) {
                // Determine percentage based on tipping rules
                var percentage;
                var bill = this.bills[i];
                if (bill < 100) {
                 percentage = 0.2;
                } else if (bill >= 100 && bill < 300) {

                // Add results to the corresponding arrays
                this.tips[i] = bill * percentage;
                this.finalValues[i] = bill + bill * percentage;

        }
    };

    function calcAverage(tips) {
        var sum = 0;

        for (var i = 0; i < tips.length; i++) {
             sum = sum + tips[i];
